<hom>

<entry>
 <form>
  <orth>bray</orth>
  <pron>breI</pron>
 </form>

  </gramGrp>
  <def>cry of an ass; sound of a trumpet.</def>
 </hom>
 <hom>
  <gramGrp>
   <pos>vt</pos>

  </gramGrp>

 </hom>
</entry>
